CREATE TABLE user (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    company VARCHAR(255),
    job VARCHAR(255),
    mail VARCHAR(255),
    phone VARCHAR(20),
    approval_date DATE,
    approver_id INT
);
